export function ReviewDetails({ reviewId }: { reviewId: string }) {
  const [review, setReview] = useState<Review | null>(null);
  const [status, setStatus] = useState<"idle" | "loading" | "ready" | "error">(
    "idle",
  );

  useEffect(() => {
    let cancelled = false;

    setStatus("loading");
    fetchReview(reviewId)
      .then((nextReview) => {

      })
      .catch(() => {

## What to review

The good version marks the request as cancelled during cleanup, so an older response cannot update state after the component has moved on.

The bad version starts a new request for every `reviewId` but allows older requests to finish later and overwrite the latest review.

## Review notes

In review, check what happens if the user navigates quickly or the component unmounts before a promise resolves. The bad version is review-worthy because it has the right dependency array but still misses the lifecycle cleanup that protects the current UI.
