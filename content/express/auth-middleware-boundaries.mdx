export async function requireUser(
  req: Request,
  res: Response,
  next: NextFunction,
) {
  // Middleware turns credentials into a small trusted user context.
  const session = await sessions.findFromHeader(req.get("authorization"));

  if (!session) {
    res.status(401).json({ error: { message: "Authentication required." } });
    return;
  }

  res.locals.user = {
    id: session.userId,
    role: session.role,
  };
  next();
}`,

## What to review

The good version authenticates in middleware and stores only the user context downstream routes need.

The bad version reparses credentials inside the route, mixing authentication, authorization, and review approval in one handler.

## Review notes

In review, auth code should have a clear boundary and consistent failure response. The bad version is review-worthy because each route can drift into its own auth logic, increasing security risk and duplication.
