export function installShutdown(server: Server, closeDatabase: () => Promise<void>) {
  async function shutdown(signal: string) {
    console.info("Received " + signal + ", shutting down");

    // Stop accepting new work before closing dependencies.
    server.close(async () => {
      await closeDatabase();
      process.exit(0);
    });

    setTimeout(() => {
      console.error("Shutdown timed out");
      process.exit(1);
    }, 10_000).unref();
  }

  process.once("SIGTERM", () => shutdown("SIGTERM"));
  process.once("SIGINT", () => shutdown("SIGINT"));
}`,

## What to review

The good version stops accepting new HTTP work, closes dependencies, and keeps a timeout in case shutdown hangs.

The bad version exits immediately, which can drop in-flight requests and leave external work half-complete.

## Review notes

In review, ask what happens during deploys, container restarts, and local Ctrl+C. The bad version is review-worthy because process signal handling is part of production correctness, not just operational polish.
